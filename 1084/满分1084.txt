void small(char *s)
{
	char b,*f;
	f = s;
	int c;
		c = 1;
		while (s[c])
		{
			while (*(s+c))
			{
				if (*s > *(s + 1))
				{
					b = *(s + 1);
					*(s + 1) = *s;
					*s = b;

				}
		
				s++;
			}
			c++;
			s = f;
		}
	
	
}
bool strless(char *a, char *b)
{
	for (; *a||*b;)
	{
		if (*a < *b)
		{
			return true;
		}
		else if(*a==*b)
		{
			a++;
			b++;
		}
		else
		{
			return false;
		}
	}
	if (*a == 0)
	{
		return true;
	}
	else
	{
		return false;
	}
}
