program ex;
var a,b:array[1..3] of longint;
i,j,n,k:longint;
begin
 read(n);
 k:=n;
 i:=0;
 while n<>0 do
 begin
  inc(i);
  a[i]:=n mod 10;
  n:=n div 10;
 end;
 for j:=3 downto 1 do
 begin
  b[j]:=1;
  for i:=1 to 3 do b[j]:=b[j]*a[j];
 end;
 if (b[1]+b[2]+b[3])=k then write('YES')
 else write('NO');
end.
