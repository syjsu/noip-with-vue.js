#include<bits/stdc++.h>
using namespace std;
int main()
{
	double a,b;
	cin>>a;
	if(a<=100000)
	{
		b=a*0.1;
	}
	else if(a>100000&&a<200000)
	{
		b=(a-100000)*0.075+100000*0.1;
	}
	else if(a>200000&&a<400000)
	{
		b=(a-200000)*0.05+(200000-100000)*0.075+100000*0.1;
	}
	else if(a>400000&&a<600000)
	{
		b=(a-400000)*0.03+(400000-200000)*0.05+(200000-100000)*0.075+100000*0.1;
	}
	else if(a>600000&&a<1000000)
	{
		b=(a-600000)*0.015+(600000-400000)*0.03+(400000-200000)*0.05+(200000-100000)*0.075+100000*0.1;
	}
	else if(a>=1000000)
	{
		b=(a-1000000)*0.01+(1000000-600000)*0.015+(600000-400000)*0.03+(400000-200000)*0.05+(200000-100000)*0.075+100000*0.1;
	}
	printf("%.3f",b);
	return 0;
}
