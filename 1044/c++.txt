#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[100001];
int b[10001];
int cz(int l,int r,int m)
{
if(r-l<=1)
return m-a[l]<=a[r]-m?a[l]:a[r];
int mid=(l+r)/2;
if(m>a[mid])
return cz(mid,r,m);
else
return cz(l,mid,m); 
}
int main(){
	int m,n,num;
scanf("%d",&n);
for(int i=1;i<=n;i++)
scanf("%d",&a[i]);
scanf("%d",&m);
for(int i=1;i<=m;i++)
{
scanf("%d",&num);
b[i]=cz(1,n,num); 
}
for(int i=1;i<=m;i++)
printf("%d\n",b[i]);
	return 0;
}
