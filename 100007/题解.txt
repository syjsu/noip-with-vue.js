//f[0][i]保存(从右往左数,下同)第i位为0的方案数
//f[1][i]保存第i位不为0的方案数
//递推式较易,详见代码
#include<cstdio>
int n,k,f[2][100];
int main()
{
	scanf("%d%d",&n,&k);
	k--;							//非0的k进制数有k-1种 
	f[1][0]=1;						//预处理第0位 
	for(int i=1;i<=n;i++)
	{
		f[0][i]=f[1][i-1];
		f[1][i]=k*(f[0][i-1]+f[1][i-1]);
	}
	printf("%d\n",f[1][n]);
}
